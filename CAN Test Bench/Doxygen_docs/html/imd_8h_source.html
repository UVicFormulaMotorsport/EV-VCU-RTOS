15// The first byte returned from a status request message will have 8 status bits and the data as the rest
16enum imd_status_bits{
17 // I have just left this as plain binary. Could also be declared by bitshifting but I think this is easier
18 // 12345678
19 Isolation_status_bit0 = 0b00000001, // IS0
20 Isolation_status_bit1 = 0b00000010, // IS1
21 Low_Battery_Voltage = 0b00000100, // LV 15V threshold (1 if below 15V)
22 High_Battery_Voltage = 0b00001000, // HV (1 if higher than Max_battery_working_voltage)
23 Exc_off = 0b00010000, // EO (Excitation pulse 0 for operating)
24 High_Uncertainty = 0b00100000, // HU (1 for greater than 5%)
25 Touch_energy_fault = 0b01000000, // EF (1 for energy exceeds 0.2J)
26 Hardware_Error = 0b10000000, // HE (1 for error)
27};

37// The MCU needs to send a message to the IMD requesting info
38// These are requests that will return status bits (defined above) & the value requested
39enum imd_status_requests{
40 // The electrical isolation is in bytes 2 & 3
41 isolation_state = 0xE0,
42
43 // Rp is the resistance from the positive of the battery to chassis
44 // Rn is the resistance from the negative of the battery to chassis
45 isolation_resistances = 0xE1,
46
47 // Cp is the capacitance from the positive of the battery to chassis
48 // Cn is the capacitance from the negative of the battery to chassis
49 // Bytes 2 & 3 are Cp, bytes 5 & 6 are Cn, in nF
50 isolation_capacitances = 0xE2,
51
52 // High voltage battery voltages to chassis
53 voltages_Vp_and_Vn = 0xE3,
54
55 // GLV battery voltage
56 battery_voltage = 0xE4,
57
58 // Error flags are a series of bits described below
59 Error_flags = 0xE5,
60
61 // The IMD monitors charge stored in the system
62 safety_touch_energy = 0xE6,
63
64 // Also monitors if it is safe to touch
65 safety_touch_current = 0xE7,
66
67 // This is just a parameter we can set
68 Max_battery_working_voltage = 0xF0,
69
70 // We can read the temperature of the board
71 Temperature = 0x80,
72};

74// If one of the error flags is set, then the harware error bit will go to 1
75enum imd_error_flags{
76 // Bits 0-6 are reserved
77 Err_temp = 0x0080, // HT high temperature (0 if temperature is below 105C)
78 Err_clock = 0x0100, // CE (0 if clock is good)
79 Err_Watchdog = 0x0200, // WD Watchdog (0 if watchdog is good)
80 Err_Vpwr = 0x0400, // V_PWR power supply (0 if power supply voltage is good)
81 Err_Vexi = 0x0800, // V_EXI Excitation voltage (0 if excitation voltage is good)
82 Err_VxR = 0x1000, // VxR (0 if both Vx1 and Vx2 are good)
83 Err_CH = 0x2000, // CH chassis connections (0 if both chassis connections are good)
84 Err_Vx1 = 0x4000, // Vx1 connection (0 if SIM101 connection to Hv+ is good)
85 Err_Vx2 = 0x8000, // Vx2 connection (0 if SIM101 connection to Hv- is good)
86};
