142static inline uv_status coniferUpdateChannel(abstract_conifer_channel* ch_ptr){
143 uint8_t src = (ch_ptr->hardware_mapping&CONIFER_CH_LOC_MASK)>>8;
144 uint32_t ecode = CONIFER_DRV_OK;

146
147 if(update_func == NULL){
148 return UV_ABORTED;
149 }
150
151 uv_status retval = update_func(ch_ptr,&ecode);//Me executing the random ass callback func
152 if(retval != UV_OK){
153 if(ch_ptr->status_control_reg & CONIFER_CH_IS_CRIT_BIT){
154 uvPanic("FATAL CONIFER FAULT",0);
155 return UV_ERROR;
156 }else{
157 return UV_WARNING;
158 }
159 }
160 return UV_OK;

173uv_status coniferInit(){
174 uint32_t init_ecode;
175
176 vTaskDelay(1);
177 //STEP 1: Init the ch_table to something normal. Set all outputs as "not in use"
178 for(int i = 0; i < FINAL_CONIFER_OUTPUT;i++){
179 abstract_conifer_channel* ch = &ch_table[i];
180 ch->hardware_mapping = 0x0700; //Makes channel as not in use and sets location as invalid
181 ch->status_control_reg = 0;
182 }
183
184
185 //STEP 2: Pull up the relevant settings into the table
186 for(int i = 0; i < conifer_params->n_ch; i++){
187 uint16_t chID = conifer_params->ch_list[i].ch;
188 ch_table[chID].hardware_mapping = conifer_params->ch_list[i].ch_dat.hardware_mapping;
189 ch_table[chID].status_control_reg = conifer_params->ch_list[i].ch_dat.status_control_reg;
190 }
191
192 vTaskDelay(1);
193 //STEP 3: Configure Driver LUTs
194
195 //Channel updaters
196 ch_updaters[UV19_PDU_CH] = u19updatePduChannel;
197 vTaskDelay(1);
198 //Current getters
199
200 //Step 4: Initialize and confirm existance of relevant external devices
201 uv_status retval = UV_OK;
202
203 //Does the PDU exist?
204 if(conifer_params->conif_flags & EXPECT_UV19PDU){
205 //retval
206 retval = initPDU(&init_ecode);
207// goto CONIFER_INIT_ERR;
208 }
209
210 if(retval != UV_OK){
211 //Bad bad not good
212 }
213
214 //STEP 5: Set system initial conditions to turn on/off the relevant channels
215 for(int i = 0; i < FINAL_CONIFER_OUTPUT; i++){
216 abstract_conifer_channel* ch_ptr = &ch_table[i];
217 if(!IS_CONIFER_CH_USED(ch_ptr)){
218 continue;
219 }
220
221 //We will update all of the conifer channels now
222 if(coniferUpdateChannel(ch_ptr) != UV_OK){
223 //Handle the error
224 }
225
226
227 }
228 return UV_OK;

236uv_status coniferEnLoadShedding(){

244 abstract_conifer_channel* ch_ptr = &ch_table[i];
245 if((ch_ptr->status_control_reg&CONIFER_CH_ALLOW_LOAD_SHED)){
246 ch_ptr->status_control_reg |= CONIFER_CH_LS_ACTIVE;
247 }else{
248 continue;

250
251 if(coniferUpdateChannel(ch_ptr) == UV_ERROR){
252 //Handle the error
253 }
254 }
255 return UV_OK;
