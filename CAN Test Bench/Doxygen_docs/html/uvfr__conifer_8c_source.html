127 uint32_t ecode = CONIFER_DRV_OK;
128 uv_status (*update_func)(abstract_conifer_channel*, uint32_t*) = ch_updaters[src];
129
130 if(update_func == NULL){
131 return UV_ABORTED;
132 }
133

135 if(retval != UV_OK){
136 switch(ecode){
137 case CONIFER_DRV_OK:
138 break;
139 case CONIFER_DRV_INVALID_HW_LOC:
140 break;
141 case CONIFER_DRV_INVALID_HW_CH_ID:
142 default:
143 break;
144 }
145 }
146 return retval;

153uv_status coniferInit(){
154 uint32_t init_ecode;
155
156 vTaskDelay(1);
157 //STEP 1: Init the ch_table to something normal. Set all outputs as "not in use"
158 for(int i = 0; i < FINAL_CONIFER_OUTPUT;i++){
159 abstract_conifer_channel* ch = &ch_table[i];
160 ch->hardware_mapping = 0x0700; //Makes channel as not in use and sets location as invalid
161 ch->status_control_reg = 0;
162 }
163
164
165 //STEP 2: Pull up the relevant settings into the table
166 for(int i = 0; i < conifer_params->n_ch; i++){
167 uint16_t chID = conifer_params->ch_list[i].ch;
168 ch_table[chID].hardware_mapping = conifer_params->ch_list[i].ch_dat.hardware_mapping;
169 ch_table[chID].status_control_reg = conifer_params->ch_list[i].ch_dat.status_control_reg;
170 }
171
172 vTaskDelay(1);
173 //STEP 3: Configure Driver LUTs
174
175 //Channel updaters
176 ch_updaters[UV19_PDU_CH] = u19updatePduChannel;
177 vTaskDelay(1);
178 //Current getters
179
180 //Step 4: Initialize and confirm existance of relevant external devices
181 uv_status retval;
182
183 //Does the PDU exist?
184 if(conifer_params->conif_flags & EXPECT_UV19PDU){
185 //retval
186 retval = initPDU(&init_ecode);
187// goto CONIFER_INIT_ERR;
188 }
189
190 if(retval != UV_OK){
191 //Bad bad not good
192 }
193
194 //STEP 5: Set system initial conditions to turn on/off the relevant channels
195 for(int i = 0; i < FINAL_CONIFER_OUTPUT; i++){
196 abstract_conifer_channel* ch_ptr = &ch_table[i];
197 if(!IS_CONIFER_CH_USED(ch_ptr)){
198 continue;
199 }
200
201 //We will update all of the conifer channels now
202 if(coniferUpdateChannel(ch_ptr) != UV_OK){
203 //Handle the error
204 }
205
206
207 }
208 return UV_OK;
209
210}
