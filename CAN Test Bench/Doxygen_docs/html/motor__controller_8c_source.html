211 void ProcessMotorControllerResponse(uv_CAN_msg* msg)
212 {
213  if (!msg || msg->dlc < 2)
214  return;
215 
216  uint8_t reg_id = msg->data[0];
217 
218  switch (reg_id) {
219  case N_actual: // SPEED_ACTUAL (0x30)
220  if (msg->dlc >= 3) {
221  int16_t speed = (int16_t)((msg->data[2] << 8) | msg->data[1]);
222  printf("SPEED_ACTUAL: %d rpm\n", speed);
223  }
224  break;
225 
226  case CURRENT_ACTUAL: // 0x31: 16-bit, little-endian
227  if (msg->dlc >= 3) {
228  int16_t current_raw = (int16_t)((msg->data[2] << 8) | msg->data[1]);
229  printf("CURRENT_ACTUAL: %d (raw units)\n", current_raw);
230  }
231  break;
232 
233  case LOGIMAP_ERRORS: // 0x82: error bitfield, little-endian
234  if (msg->dlc >= 3) {
235  MotorControllerErrorHandler_16bitLE(&msg->data[1], 2);
236  }
237  break;
238 
239  case LOGIMAP_IO: // 0x83: I/O status, 16-bit, little-endian
240  if (msg->dlc >= 3) {
241  uint16_t io_flags = (uint16_t)((msg->data[2] << 8) | msg->data[1]);
242  printf("LOGIMAP_IO flags: 0x%04X\n", io_flags);
243  }
244  break;
245 
246  case POS_ACTUAL: // 0x86: 32-bit value, little-endian
247  if (msg->dlc >= 5) {
248  int32_t pos = (int32_t)((msg->data[4] << 24) |
249  (msg->data[3] << 16) |

253  }
254  break;
