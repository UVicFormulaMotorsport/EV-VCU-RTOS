69uint16_t sendTorqueToMotorController(float T_filtered)
70{
71 if (T_filtered < 0.0f)
72 T_filtered = 0.0f;
73 if (T_filtered > 256.0f)
74 T_filtered = 256.0f;
75
76 //uint16_t torque_cmd = (uint16_t) T_filtered;
77
78 //scale to bamocar requirements of 32760
79 float T_MAX = 230.0f;
80 uint16_t torque_cmd = (uint16_t)((T_filtered / T_MAX) * 32760.0f);
81
82 //if T_filtered is 115 nm --> 1680 for bamocar
83 //(115 / 230) * 32760 = 16380 ≈ 0x3FFC
84
85 static uv_CAN_msg torque_msg;
86 memset(&torque_msg, 0, sizeof(torque_msg));
87
88 torque_msg.msg_id = mc_settings->can_id_tx;
89 torque_msg.dlc = 3;
90 // Use the N_set command (from your enum motor_controller_speed_parameters)
91 torque_msg.data[0] = N_set; //speed comand
92 //torque_msg.data[0] = M_Set; //torque command
93 // Little-endian: LSB first then MSB
94 torque_msg.data[1] = (uint8_t)(torque_cmd & 0xFF);
95 torque_msg.data[2] = (uint8_t)((torque_cmd >> 8) & 0xFF);
96 torque_msg.flags = mc_settings->mc_bus;
97
98 if (uvSendCanMSG(&torque_msg) != UV_OK) {
99 uvPanic("Failed to send torque command", 0);
100 return 1;
101 }
102 return 0;

324void ProcessMotorControllerResponse(uv_CAN_msg* msg){
325
326 //every incoming mc message gets stored for use later
327 memcpy(&last_mc_response, msg, sizeof(uv_CAN_msg));
328
329 if (!msg || msg->dlc < 2)
330 return;
331
332 uint8_t reg_id = msg->data[0];
333
334 switch (reg_id) {
335 case N_actual: // SPEED_ACTUAL (0x30)
336 if (msg->dlc >= 3) {
337 int16_t speed = (int16_t)((msg->data[2] << 8) | msg->data[1]);
338 mc_speed_rpm = (int16_t)((msg->data[2] << 8) | msg->data[1]); //cyclic
339 }
340 break;
341
342 case CURRENT_ACTUAL: // 0x31: 16-bit, little-endian
343 if (msg->dlc >= 3) {
344 int16_t current_raw = (int16_t)((msg->data[2] << 8) | msg->data[1]);
345 mc_current = (int16_t)((msg->data[2] << 8) | msg->data[1]); //cyclic
346 }
347 break;
348
349 case LOGIMAP_ERRORS: // 0x82: error bitfield, little-endian
350 if (msg->dlc >= 3) {
351 MotorControllerErrorHandler_16bitLE(&msg->data[1], 2);
352 }
353 break;
354
355 case LOGIMAP_IO: // 0x83: I/O status, 16-bit, little-endian
356 if (msg->dlc >= 3) {
357 uint16_t io_flags = (uint16_t)((msg->data[2] << 8) | msg->data[1]);
358 }
359 break;
360
361 case POS_ACTUAL: // 0x86: 32-bit value, little-endian
362 if (msg->dlc >= 5) {
363 int32_t pos = (int32_t)((msg->data[4] << 24) |
364 (msg->data[3] << 16) |
365 (msg->data[2] << 8) |
366 msg->data[1]);
367 }
368 break;
369
370 case motor_controller_errors_warnings:
371 // For error/warning responses using this register, assume a 16-bit field
372 if (msg->dlc >= 3) {
373 MotorControllerErrorHandler_16bitLE(&msg->data[1], 2);
374 }
375 break;
376 case M_out: //0xA0: actual active current scaled
377 mc_torque_cmd = (int16_t)((msg->data[2] << 8) | msg->data[1]); //cyclic
378 break;
379
380 case motor_temperature: //0x49: motor temperature
381 mc_motor_temp = (int16_t)((msg->data[2] << 8) | msg->data[1]); //cyclic
382 break;
383
384 case igbt_temperature: //0x4A: igbt temperature
385 mc_igbt_temp = (int16_t)((msg->data[2] << 8) | msg->data[1]); //cyclic
386 break;
387
388 default:
389 // Handle other responses as needed or call a default parser.
390 break;
391 }
392}

407void MC_EnableCyclicSpeedTransmission(uint8_t interval_ms)
408{
409 if (interval_ms < 1 || interval_ms > 254) return;
410
411 uint8_t regs[] = {
412 N_actual, // 0x30 — Actual Speed
413 CURRENT_ACTUAL, // 0x69 — Actual Current
414 M_out, // 0xA0 — Actual Active Current Scaled
415 motor_temperature, // 0x49 — Current Motor temperature
416 igbt_temperature, // 0x4A — IGBT temperature
417 max_motor_temp, //0xa3 - max motor temp
418 warning_motor_temp, //0xa2 - warning motor temp
419 LOGIMAP_ERRORS, // 0x8F — ERROR BIT map
420 };
421
422 for (int i = 0; i < sizeof(regs); i++) {
423 uv_CAN_msg tx;
424 memset(&tx, 0, sizeof(tx));
425
426 tx.msg_id = mc_settings->can_id_tx;
427 tx.dlc = 3;
428 tx.data[0] = 0x3D; // Command: Enable cyclic read
429 tx.data[1] = regs[i]; // Target register
430 tx.data[2] = interval_ms; // Repeating time (1–254 ms)
431 tx.flags = mc_settings->mc_bus;
432
433 uvSendCanMSG(&tx);
434 vTaskDelay(pdMS_TO_TICKS(5)); // delay between messages
435 }
436}

548void MC_Shutdown(void)
549{
550 // 1. Stop all cyclic transmission
551 MC_EnableCyclicSpeedTransmission(0xFF); // 0xFF disables periodic streaming
552 vTaskDelay(pdMS_TO_TICKS(10));
553
554 // 2. Set speed or torque to zero
555 // Use one of these depending on your control mode:
556 // --- If using speed mode:

562
