160void handleIncomingLaptopMsg(uv_CAN_msg* msg) PRIVILEGED_FUNCTION{
161 uint8_t cmd_byte = msg->data[0];
162 static uv_CAN_msg blank_msg = {
163 .msg_id = 0x420,
164 .flags = 0x00,
165
166 .dlc = 8
167 };
168
169 last_contact_with_laptop = xTaskGetTickCount();
170
171
172 switch(cmd_byte){
173 case LAPTOP_HANDSHAKE:
174 is_laptop_connected = true;
175 last_contact_with_laptop = xTaskGetTickCount();
176
177 blank_msg.dlc = 8;
178 blank_msg.data[0] = 0x01;
179 serializeSmallE32(blank_msg.data,last_contact_with_laptop,1);
180 blank_msg.data[5] = FIRMWARE_MAJOR_RELEASE;
181 blank_msg.data[6] = FIRMWARE_MINOR_RELEASE;
182 blank_msg.data[7] = FIRMWARE_PATCH_NUM;
183
184
185 uvSendCanMSG(&blank_msg);
186 break;
187 case ENTER_PROGRAMMING_MODE:
188
189 if(vehicle_state == UV_READY){
190
191 //Relevant Ack. frame sent upon succesful startup of programmer task
192
193 changeVehicleState(PROGRAMMING);
194 }else{
195 //should not be programming if in either of these states
196 uvSendCanMSG(&vcu_ack_failed_msg);
197 }
198 break;
199 case REQUEST_VCU_STATUS:
200 //TODO Respond with VCU status
201
202 if(uvTransmitVehicleStatus()!=UV_OK){
203 //Handle this error.
204 }
205
206 break;
207 case GENERIC_ACK:
208 //Could come at any time, however here we are
209 break;
210 case REQUEST_ALL_JOURNAL_ENTRIES:
211 flushLogsToCAN();
212 break;
213 case SET_SPECIFIC_PARAM ... END_OF_SPECIFIC_PARAMS:
214
215 case CLEAR_FAULTS:
216 case REQUEST_ALL_SETTINGS:
217 case REQUEST_JOURNAL_ENTRIES_BY_TIME:
218 case REQUEST_SPECIFIC_SETTING:
219 case SAVE_AND_APPLY_NEW_SETTINGS:
220 case DISCARD_NEW_SETTINGS:
221 case DISCARD_NEW_SETTINGS_AND_EXIT:
222 case FORCE_RESTORE_FACTORY_DEFAULT:
223 //We need to be in programming mode for these.
224 if(vehicle_state == PROGRAMMING){
225 if(settings_queue != NULL){
226 if(xQueueSend(settings_queue,msg,0) != pdTRUE){
227 //error
228 uvSendCanMSG(&vcu_ack_failed_msg);
229 uvPanic("",0);
230 }
231 }
232 }else{
233 //send negative ACK
234 uvSendCanMSG(&vcu_ack_failed_msg);
235 }
236 //otherwise we will ignore
237 break;
238 default:
239
240 //cmd_byte we do not recognise. We assme the laptop is geeked, and move on
241
242 break;
243 }
244}
