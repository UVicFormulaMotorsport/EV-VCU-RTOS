487uv_status uvIsPTRValid(void* ptr){
488 if(ptr == NULL){
489 return UV_WARNING;
490 }
491 uint32_t pval = (uint32_t)ptr;
492
493 //bool is_valid = false;
494
495 if(pval < 0x000FFFFF){ //Aliased to FLASH, systmem or SRAM
496 return UV_OK;
497 }
498
499 if((pval > 0x08000000)&& (pval < 0x080FFFFF)){ //Flash be like
500 return UV_OK;
501 }
502
503 if((pval > 0x10000000)&&(pval < 0x1000FFFF)){ //CCM Data RAM
504 return UV_OK;
505 }
506
507 if((pval > 0x1FFF0000)&&(pval < 0x1FFF7A0F)){ //System memory + OTP
508 return UV_OK;
509 }
510
511 if((pval > 0x1FFFC000)&&(pval < 0x1FFFC007)){ //option bytes (should these be user accessable under any circumstances?)
512 return UV_WARNING;
513 }
514
515 if((pval > 0x20000000)&&(pval < 0x2001FFFF)){ //SRAM :)
516 return UV_OK;
517 }
518
519 if((pval > 0x40000000)&&(pval < 0x40007FFF)){ //APB1
520 return UV_OK;
521 }
522
523 if((pval > 0x40010000)&&(pval < 0x400157FF)){ //APB2
524 return UV_OK;
525 }
526
527 if((pval > 0x40020000)&&(pval < 0x4007FFFF)){ //AHB1
528 return UV_OK;
529 }
530
531 if((pval > 0x50000000)&&(pval < 0x50060BFF)){//AHB2
532 return UV_OK;
533 }
534
535 if((pval > 0x60000000)&&(pval < 0xA0000FFF)){//AHB3
536 return UV_OK;
537 }
538
539 if((pval > 0xE0000000)&&(pval < 0xE00FFFFF)){//
540 return UV_OK;
541 }

543 return UV_ERROR;
