76void handleCANbusError(const CAN_HandleTypeDef* hcan, const uint32_t err_to_ignore){
77 is_can_ok = 0;
78 if(hcan == NULL){
79 uvPanic("null can handle",0);
80 return;
81 }
82
83
84 uint32_t errcode = HAL_CAN_GetError(hcan);
85
86 if(errcode & err_to_ignore){
87 return;
88 }
89
90 if(errcode == HAL_CAN_ERROR_NONE){
91 return;
92 }else if(errcode & HAL_CAN_ERROR_EWG){ //protocol error
93 uvPanic("CAN protocol error",0);
94 }else if(errcode & HAL_CAN_ERROR_EPV){ //passive
95 uvPanic("CAN passive error",0);
96 }else if(errcode & HAL_CAN_ERROR_BOF){ //Bus-off error
97 uvPanic("CAN Bus off error",0);
98 }else if(errcode & HAL_CAN_ERROR_STF){ //Stuff error
99 uvPanic("CAN Stuff error",0);
100 }else if(errcode & HAL_CAN_ERROR_FOR){ //Form error
101 uvPanic("CAN Form error",0);
102 }else if(errcode & HAL_CAN_ERROR_ACK){ //Acknowledgement error
103 uvPanic("CAN Ack error",0);
104 }else if(errcode & HAL_CAN_ERROR_BR){ //bit recessive
105 uvPanic("CAN Recessive Bit error",0);
106 }else if(errcode & HAL_CAN_ERROR_BD){ //bit dominant
107 uvPanic("CAN Dominant Bit error",0);
108 }else if(errcode & HAL_CAN_ERROR_CRC){//cyclic redundancy check
109 uvPanic("CAN CRC Failed",0);
110 }else if(errcode & HAL_CAN_ERROR_RX_FOV0){//overrun rx fifo0
111 uvPanic("CAN RX FIFO0",0);
112 }else if(errcode & HAL_CAN_ERROR_RX_FOV1){//overrun rx fifo1
113 uvPanic("CAN RX FIFO1",0);
114 }else if(errcode & HAL_CAN_ERROR_TX_ALST0){//tx mailbox 0 arbitration lost
115 uvPanic("CAN0 arbitration",0);
116 }else if(errcode & HAL_CAN_ERROR_TX_TERR0){//tx mailbox 0 transmit error
117 uvPanic("CAN0 transmit err",0);
118 }else if(errcode & HAL_CAN_ERROR_TX_ALST1){//tx mailbox 1 arbitration lost error
119 uvPanic("CAN1 arbitration",0);
120 }else if(errcode & HAL_CAN_ERROR_TX_TERR1){//tx mailbox 1 transmit error
121 uvPanic("CAN1 transmit err",0);
122 }else if(errcode & HAL_CAN_ERROR_TX_ALST2){//tx mailbox 2 arbitration lost error
123 uvPanic("CAN2 arbitration",0);
124 }else if(errcode & HAL_CAN_ERROR_TX_TERR2){//tx mailbox 2 transmit error
125 uvPanic("CAN2 transmit err",0);
126 }else if(errcode & HAL_CAN_ERROR_TIMEOUT){//timeout
127 uvPanic("CAN timeout",0);
128 }else if(errcode & HAL_CAN_ERROR_NOT_INITIALIZED){//is not initialized
129 uvPanic("CAN not init",0);
130 }else if(errcode & HAL_CAN_ERROR_NOT_READY){//Not ready
131 uvPanic("CAN_not_ready",0);
132 }else if(errcode & HAL_CAN_ERROR_NOT_STARTED){//not started
133 uvPanic("HAL_CAN_NOT_STARTED",0);
134 }else if(errcode & HAL_CAN_ERROR_PARAM){//Param
135 uvPanic("CAN Param",0);
136 }else if(errcode & HAL_CAN_ERROR_INVALID_CALLBACK){//invalid callback
137 uvPanic("Invalid_callback",0);
138 }else if(errcode & HAL_CAN_ERROR_INTERNAL){//internal error
139 uvPanic("HAL_internal",0);

141 //no clue how we got here
