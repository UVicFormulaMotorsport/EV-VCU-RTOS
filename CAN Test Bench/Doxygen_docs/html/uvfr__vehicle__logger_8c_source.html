97void logVehicleFault(fault_event_type_e type,
98 uv_task_info* task,
99 const char* msg,
100 const char* file,
101 int line,
102 const char* func,
103 bool is_panic)
104{
105 // Mark this fault type as "occurred"
106 system_error_flags |= (1U << type);
107
108 // Write into next log slot
109 vehicle_log_entry* entry = &log_buffer[log_write_index];
110
111 entry->type = type;
112 entry->msg = msg;
113 entry->file = file;
114 entry->line = line;
115 entry->func = func;
116 entry->is_panic = is_panic;
117 entry->timestamp = xTaskGetTickCount();
118 entry->vehicle_state = vehicle_state;
119
120 // Add task info (if task is valid)
121 if (task != NULL) {
122 entry->task_name = task->task_name;
123 entry->task_id = task->task_id;
124 entry->task_state = task->task_state;
125 } else {
126 entry->task_name = "None";
127 entry->task_id = 0xFF;
128 entry->task_state = UV_TASK_NOT_STARTED;
129 }
130
131 // Move to next slot
132 log_write_index++;
133 if (log_write_index >= MAX_LOG_ENTRIES) {
134 log_write_index = 0;
135 has_wrapped = true;
136 }
137}

155void flushLogsToCAN(void)
156{
157 uint8_t count = getLogCount();
158
159 for (uint8_t i = 0; i < count; i++)
160 {
161 const vehicle_log_entry* entry = getLogEntry(i);
162 if (entry == NULL) continue;
163
164 uv_CAN_msg msg;
165 msg.msg_id = LOG_DUMP_CAN_ID;
166 msg.dlc = 8;
167 msg.flags = 0x00;
168
169 msg.data[0] = (uint8_t)entry->type;
170 msg.data[1] = (uint8_t)(entry->timestamp & 0xFF);
171 msg.data[2] = (uint8_t)((entry->timestamp >> 8) & 0xFF);
172 msg.data[3] = (uint8_t)((entry->timestamp >> 16) & 0xFF);
173 msg.data[4] = (uint8_t)((entry->timestamp >> 24) & 0xFF);
174 msg.data[5] = (uint8_t)entry->task_id;
175 msg.data[6] = (uint8_t)entry->is_panic;
176 msg.data[7] = (uint8_t)entry->vehicle_state;
177 msg.data[8] = 0;
178 msg.data[9] = 0;
179
180 uvSendCanMSG(&msg); // <-- Use your existing CAN transmit wrapper
181 osDelay(2); // small gap between packets
182 }
